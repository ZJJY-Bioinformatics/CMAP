<script>
const log = console.log.bind(console)
const e = (sel) => document.querySelector(sel)
const isMobile = () => document.body.clientWidth < 1024

const getUrlRedirect = () => {
if (isMobile()) {
return 'https://js.design/ai?code'
}
return 'https://js.design/ai'
}

const createStyleHide = function () {
// 创键特殊
const styleNode = document.createElement('style');
// 设置 style 节点的样式
styleNode.textContent = `
#topBar, #homeBar {
display: none;
}`;
return styleNode
}

const cloneStyles = function (styles) {
for (let index = 0; index < styles.length; index++) {
const cloneNode = styles[index].cloneNode(true);
document.body.appendChild(cloneNode);
}
}

const adjustContentNode = function (node) {
if (!node || isMobile()) return;
const pNode = node.parentElement;
const height = window.getComputedStyle(pNode).height;
pNode.style.height = height;
}

const cloneBottomTabBar = function (node) {
if (!node) return;
const cloneNode = node.cloneNode(true);
const parent = node.parentElement;
const removed = parent.removeChild(node);
const transformStyle = isMobile()
? `
transform-origin: left bottom;
transform: scale(${calScaleValue()});`
: 'margin: 0 auto;';
cloneNode.style = `
position: fixed;
bottom: 0;
top: unset;
left: 0;
right: 0;
${transformStyle}
`;
cloneNode.zIndex = -1;
const c = e('#content');
if (c && c.parentElement) {
c.parentElement.appendChild(cloneNode);
}
}

const changeMarginBottom = function (node, shadowRoot) {
const bottomTabBar = document.getElementById('BottomTabBar');
if (!bottomTabBar || !isMobile()) return;
const height = parseInt(window.getComputedStyle(bottomTabBar).height);

shadowRoot.querySelector('div').style.marginBottom = height + 'px';
}

const getUrlField = (key) => {
let url = location.search.slice(1).split('&')
let urlForm = url.reduce((pre, cur) => {
let list = cur.split('=')
pre[list[0]] = list[1]
return pre
}, {})
return urlForm[key];
}

const init = () => {
let id = getUrlField('id');
let content = document.getElementById("content");
if (id) {
let node = document.getElementById('js-frame');
node.src = `https://ds.js.design/elements/${id}`;
node.update();

const shadowRoot = node.shadowRoot;
const styleNode = createStyleHide();
shadowRoot.insertBefore(styleNode, shadowRoot.querySelector('div'));

window.onload = function () {
const w = window.jsdD2CfirstFrameWidth
if (!w || w === 390) {
setContentScale()
const el = e('#content');
if (el) {
el.style.width = '390px';
}
}
const styles = shadowRoot.querySelectorAll('style');
const node = shadowRoot.getElementById('BottomTabBar');
if (node) {
adjustContentNode(node);
cloneBottomTabBar(node);
cloneStyles(styles);
changeMarginBottom(node, shadowRoot);
}

if (isMobile()) {
const t = e('#tip_mobile')
if (!node) {
t.style.bottom = `24px`
}
const c = window.getComputedStyle(t)
t.style.bottom = `calc(${c.bottom} * ${calScaleValue()})`

// 如果是大于1440的画板，代表是ai生成的web页面，不采用自动布局自适应，设置body宽度，使得可以上下左右可以滚动
if (w >= 1440) {
document.body.style.width = `${w}px`
}
}
}
} else {
let node = document.createElement('div');
node.innerText = '请输入id字段，例如：?id=xxxx';
content.appendChild(node);
}
}

const calScaleValue = () => {
const w = document.body.clientWidth
return w / 390
}

const setContentScale = () => {
const d = e('#content')
if (!d) return;
if (isMobile()) {
d.style = `
transform-origin: left top;
transform: scale(${calScaleValue()});
overflow-x: hidden;
`
} else {
d.style = `
margin: 0 auto;
`
}
}

const setBodyHeight = () => {
document.body.style.height = 0 + 'px';
}

const updateTitle = () => {
let t = getUrlField('title');
if (t) {
document.title = decodeURIComponent(t)
}
}

const showTip = () => {
let t = null
if (isMobile()) {
t = e('#tip_mobile')
} else {
t = e('#tip_pc')
}
if (t) {
t.style.display = 'flex'
}
}

const goToAi = () => {
// window.location.href = getUrlRedirect()
}

const handleGoToAi = () => {
window.addEventListener('load', function () {
const jsFrame = e('#js-frame')
if (jsFrame && jsFrame.shadowRoot) {
const shadowRoot = jsFrame.shadowRoot;
const d = shadowRoot.querySelector('div')
if (d && d.hasChildNodes()) {
showTip()
} else {
goToAi()
}
} else {
goToAi()
}
})
}

const __main = () => {
init()
updateTitle()
handleGoToAi()
if (isMobile()) {
setBodyHeight()
}
}

__main()
</script>
